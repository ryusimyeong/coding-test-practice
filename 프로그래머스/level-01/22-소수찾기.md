## 구현

시간 초과 풀이

```js
const solution = n => {
    let answer = 0;
    const nArr = Array(n-1).fill(0).map((v, i) => i+2);
    
    for (let number of nArr) {
        const arr = [];
        let isPrime = true;
        
        for (let i = 1; i <= number; i++) {
            if (number % i === 0) arr.push(i);
        }
        
        for (let i = 2; i <= Math.floor(Math.sqrt(number)); i++) {
            if (arr.includes(i)) {
                isPrime = false;
                break;
            }
        }
        if (isPrime) answer++;
    }
    return answer;
};
```
